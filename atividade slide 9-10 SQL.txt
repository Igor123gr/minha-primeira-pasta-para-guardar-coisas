--------------------- 2
SELECT AVG(contagem_dia) AS media_emprestimos
FROM (
    SELECT COUNT(*) AS contagem_dia
    FROM emprestimo
    WHERE DATE(datemp) = '2025-02-10'
) AS sub;

--------------------- 7
SELECT DATE(reserva.datres) AS data, COUNT(*) AS quantidade_reservas
FROM reserva
GROUP BY DATE(reserva.datres)
HAVING COUNT(*) = (
    SELECT MAX(total) FROM (
        SELECT COUNT(*) AS total
        FROM reserva
        GROUP BY DATE(datres)
    ) AS sub
);

--------------------- 10
SELECT livro.titliv, COUNT(*) AS quantidade_reservas
FROM livro
JOIN reserva ON reserva.codliv = livro.codliv
GROUP BY livro.codliv, livro.titliv
ORDER BY quantidade_reservas DESC
LIMIT 1;
